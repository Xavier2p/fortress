12/// Structure of the database.
13/// Contains a check if the database is readable, and the data.
14#[derive(Serialize, Deserialize)]
15struct DatabaseWrapper {
16    /// Password check to ensure the database is readable.
17    _pwcheck: String,
18
19    /// Actual data.
20    entries: Vec<PasswordEntry>,
21}

23/// Derive a 256-bit key from `password` using `Argon2id`
24/// ## Parameters:
25/// - `password`: The password to derive the key from.
26/// - `salt`: The
27/// ## Returns:
28/// A result of the key or a [`FortressError`]
29fn derive_key(password: &str, salt: &[u8; 32]) -> Result<[u8; 32], FortressError> {
30    let params = Params::new(65536, 3, 4, Some(32)).map_err(|_| FortressError::EncryptionFailed)?;
31    let argon2 = Argon2::new(argon2::Algorithm::Argon2id, argon2::Version::V0x13, params);
32    let mut key = [0u8; 32];
33    argon2
34        .hash_password_into(password.as_bytes(), salt, &mut key)
35        .map_err(|_| FortressError::EncryptionFailed)?;
36    Ok(key)
37}

39/// Encrypt the password database
40/// ## Process
41/// 1. Create a wrapper with password check.
42/// 2. Serialize to JSON
43/// 3. Generate random salt
44/// 4. Derive key from password
45/// 5. Create cipher and generate nonce
46/// 6. Encrypt
47/// 7. Build final format: [Salt: 32 bytes][Nonce: 12 bytes][Encrypted Data + Auth Tag]
48/// ## Parameters
49/// - `entries`: All entries to save
50/// - `master_password`: The password to encrypt data.
51/// ## Returns
52/// A result of the encrypted database as bytes or a [`FortressError`].
53#[allow(deprecated)]
54pub fn encrypt_database(
55    entries: &[PasswordEntry],
56    master_password: &str,
57) -> Result<Vec<u8>, FortressError> {
58    let wrapper = DatabaseWrapper {
59        _pwcheck: "valid".to_string(),
60        entries: entries.to_vec(),
61    };
62
63    let json_data = serde_json::to_string(&wrapper)?;
64    let plaintext = json_data.as_bytes();
65
66    let mut salt_32 = [0u8; 32];
67    rand::rng().fill_bytes(&mut salt_32);
68
69    let key_bytes = derive_key(master_password, &salt_32)?;
70    let key = Key::<Aes256Gcm>::from_slice(&key_bytes);
71
72    let cipher = Aes256Gcm::new(key);
73    let mut nonce_bytes = [0u8; 12];
74    rand::rng().fill_bytes(&mut nonce_bytes);
75    let nonce = Nonce::from_slice(&nonce_bytes);
76
77    let ciphertext = cipher
78        .encrypt(nonce, plaintext)
79        .map_err(|_| FortressError::EncryptionFailed)?;
80
81    let mut result = Vec::with_capacity(32 + 12 + ciphertext.len());
82    result.extend_from_slice(&salt_32);
83    result.extend_from_slice(&nonce_bytes);
84    result.extend_from_slice(&ciphertext);
85
86    Ok(result)
87}

89/// Decrypt the password database
90/// ## Process
91/// 1. Check minimum file size (32 + 12 + 16 = 60 bytes minimum)
92/// 2. Extract components: [Salt: 32 bytes][Nonce: 12 bytes][Encrypted Data + Auth Tag]
93/// 3. Derive key from password
94/// 4. Create cipher and nonce
95/// 5. Decrypt
96/// 6. Parse JSON
97/// 7. Verify password check
98/// ## Parameters
99/// - `encrypted_data`: The encrypted database as bytes.
100/// - `master_password`: The password to decrypt data.
101/// ## Returns
102/// A result of the decrypted entries or a [`FortressError`].
103#[allow(deprecated)]
104pub fn decrypt_database(
105    encrypted_data: &[u8],
106    master_password: &str,
107) -> Result<Vec<PasswordEntry>, FortressError> {
108    if encrypted_data.len() < 60 {
109        return Err(FortressError::CorruptedVault);
110    }
111
112    let salt: [u8; 32] = encrypted_data[0..32]
113        .try_into()
114        .map_err(|_| FortressError::CorruptedVault)?;
115    let nonce_bytes: [u8; 12] = encrypted_data[32..44]
116        .try_into()
117        .map_err(|_| FortressError::CorruptedVault)?;
118    let ciphertext = &encrypted_data[44..];
119
120    let key_bytes = derive_key(master_password, &salt)?;
121    let key = Key::<Aes256Gcm>::from_slice(&key_bytes);
122
123    let cipher = Aes256Gcm::new(key);
124    let nonce = Nonce::from_slice(&nonce_bytes);
125
126    let plaintext = cipher
127        .decrypt(nonce, ciphertext)
128        .map_err(|_| FortressError::InvalidMasterPassword)?;
129
130    let json_str = std::str::from_utf8(&plaintext).map_err(|_| FortressError::CorruptedVault)?;
131
132    let wrapper: DatabaseWrapper = serde_json::from_str(json_str)?;
133
134    if wrapper._pwcheck != "valid" {
135        return Err(FortressError::InvalidMasterPassword);
136    }
137
138    Ok(wrapper.entries)
139}

146    #[test]
147    fn test_encrypt_decrypt_cycle() {
148        let entries = vec![
149            PasswordEntry {
150                identifier: "Gmail".to_string(),
151                username: "user@gmail.com".to_string(),
152                password: "super_secret_123".to_string(),
153            },
154            PasswordEntry {
155                identifier: "GitHub".to_string(),
156                username: "developer".to_string(),
157                password: "github_token_456".to_string(),
158            },
159        ];
160
161        let master_password = "my_master_password";
162
163        // Encrypt
164        let encrypted =
165            encrypt_database(&entries, master_password).expect("Encryption should succeed");
166
167        // Decrypt
168        let decrypted =
169            decrypt_database(&encrypted, master_password).expect("Decryption should succeed");
170
171        assert_eq!(entries.len(), decrypted.len());
172        assert_eq!(entries[0].identifier, decrypted[0].identifier);
173        assert_eq!(entries[0].username, decrypted[0].username);
174        assert_eq!(entries[0].password, decrypted[0].password);
175    }
176
177    #[test]
178    fn test_wrong_password() {
179        let entries = vec![PasswordEntry {

182            password: "test123".to_string(),
183        }];
184
185        let encrypted =

187

190    }
