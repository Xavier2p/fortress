1//! CLI related structs and functions.
2use clap::{Parser, Subcommand};
3
4/// The CLI context.
5#[derive(Parser)]
6#[command(name = "frt-rs", version, about, long_about = None)]
7#[command(propagate_version = true)]
8pub struct Cli {
9    /// The command to run
10    #[command(subcommand)]
11    pub command: Option<Commands>,
12
13    /// The input file path
14    #[arg(short, long, value_name = "PATH", default_value = "/tmp/vault.frt")]
15    pub file: Option<String>,
16
17    /// Path to a file to write logs to
18    #[arg(long, value_name = "PATH", default_value = "/tmp/fortress.log")]
19    pub log_file: Option<String>,
20}
21
22/// The different commands that can be run.
23#[derive(Subcommand)]
24pub enum Commands {
25    /// Create a new vault
26    Create {
27        /// Overwrite the vault if it already exists
28        #[arg(short, long)]
29        force: bool,
30    },
31
32    /// Copy the password of the desired identifier
33    Copy {
34        /// The identifier of the entry
35        identifier: String,
36    },
37
38    /// View the password of the desired identifier
39    View {
40        /// The identifier of the entry
41        identifier: String,
42    },
43
44    /// Remove an entry from the vault
45    Remove {
46        /// The identifier of the entry to remove
47        identifier: String,
48    },
49
50    /// Add a new entry to the vault. If no one of the password methods is provided,
51    /// the password will be the content of the clipboard.
52    #[command(arg_required_else_help = true)]
53    Add {
54        /// The identifier for the entry
55        identifier: String,
56
57        /// The username or email address for the entry
58        #[arg(short, long)]
59        username: Option<String>,
60
61        /// Generate a new password. Mutually exclusive with 'password'
62        #[arg(short, long, conflicts_with = "password")]
63        generate: bool,
64
65        /// Direct password input. Mutually exclusive with 'generate'
66        #[arg(short, long, conflicts_with = "generate")]
67        password: Option<String>,
68    },
69    /// List all entries in the vault
70    List {},
71}
