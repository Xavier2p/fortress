1//! Some structs used throughout the program.
2use serde::{Deserialize, Serialize};
3use std::fmt;
4
5/// A single entry in the vault.
6#[derive(Serialize, Deserialize, Clone)]
7pub struct PasswordEntry {
8    /// The identifier for the entry. Can be see as the path to the entry.
9    pub identifier: String,
10    /// The username for the entry.
11    pub username: String,
12    /// The password for the entry.
13    pub password: String,
14}
15
16/// Display the entry in a readable format.
17impl fmt::Display for PasswordEntry {
18    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
19        write!(
20            f,
21            "{} ({}): '{}'",
22            self.identifier,
23            self.username,
24            "*".repeat(self.password.len())
25        )
26    }
27}
28
29/// The context of the program.
30#[derive(Clone)]
31pub struct GeneralArgs {
32    /// Enable verbose output
33    pub verbose: bool,
34    /// The input file path
35    pub file: String,
36    /// The master password
37    pub password: String,
38}
39
40/// Function to use the program context.
41impl GeneralArgs {
42    /// Create a new context.
43    pub fn new(verbose: bool, file: String, password: String) -> Self {
44        GeneralArgs {
45            verbose,
46            file,
47            password,
48        }
49    }
50}

52#[cfg(test)]

56    #[test]

58        let entry = PasswordEntry {
59            identifier: "id".to_string(),
60            username: "user".to_string(),
61            password: "secret".to_string(),
62        };
63        let display = format!("{}", entry);
64        assert!(display.contains("id"));
65        assert!(display.contains("user"));
66        assert!(display.contains("******"));
67    }
68
69    #[test]
70    fn test_general_args_new() {
71        let args = GeneralArgs::new(true, "file".to_string(), "pw".to_string());
72        assert!(args.verbose);
73        assert_eq!(args.file, "file");
74        assert_eq!(args.password, "pw");
