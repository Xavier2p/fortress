1//! List all entries in the vault.

5/// List all entries in the vault.
6/// All the entries will be printed with the format defined.
7/// ## Parameters:
8/// - `args`: The context of the program
9/// ## Returns:
10/// A result of nothing or a [`FortressError`]
11pub fn list(args: GeneralArgs) -> Result<(), FortressError> {
12    let decrypted = helpers::load_vault(args);
13    match decrypted {
14        Ok(decrypted) => {
15            println!("[");
16            decrypted.iter().for_each(|item| println!("\t{}", item));
17            println!("]");
18            Ok(())
19        }
20        Err(e) => Err(e),
21    }
22}
